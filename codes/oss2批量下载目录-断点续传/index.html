    import os
import oss2

oss_config = {
    'endpoint': 'oss-cn-shanghai.aliyuncs.com',
    'bucket': 'bucketname',
    'accessKeyId': 'accessKeyId',
    'accessKeySecret': 'accessKeySecret',
}

oss_prefix = 'oss_prefix'
download_local_save_prefix = "./"


auth = oss2.Auth(oss_config['accessKeyId'], oss_config['accessKeySecret'])
bucket = oss2.Bucket(auth, oss_config['endpoint'], oss_config['bucket'])

cache_file_path = download_local_save_prefix + "cache_file." + oss_prefix.replace("/", ".")
isExists = os.path.exists(cache_file_path)
if not isExists:
    with open(cache_file_path, "w") as fw:
        num = 0
        dirs = {}
        for b in oss2.ObjectIterator(bucket, prefix=oss_prefix):
            base_dir = os.path.dirname(download_local_save_prefix + b.key)
            if base_dir not in dirs:
                isExists = os.path.exists(base_dir)
                if not isExists:
                    os.makedirs(base_dir)
                    print(base_dir, 'create success')
            if num%100 == 0:
                pass
                # print(num)
            fw.write(b.key+"\n")
            # bucket.get_object_to_file(b.key, download_local_save_prefix + b.key)
            num += 1
else:
    with open(cache_file_path, "r") as fr:
        s = fr.read().split("\n")
        for line in s:
            if line!="":
                isExists = os.path.exists(line)
                if not isExists:
                    print("download: " + line)
                    bucket.get_object_to_file(line, download_local_save_prefix + line)
