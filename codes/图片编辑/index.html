    图片合并

    def merge(image1, image2):
    """
    两张图片等权重合并
    :param image1:
    :param image2: 
    :return:
    """
    rows1 = image1.shape[0]
    cols1 = image1.shape[1]
    channels = image1.shape[2]
    for i in range(rows1):
        for j in range(cols1):
            for channel in range(channels):
                image1[i, j][channel] = int(image1[i, j][channel]/2 
                                            + image2[i, j][channel]/2)


def add_icon(image, icon_image):
    """
    在image大图上加图标, icon_image为rgb小图
    :param image:
    :param icon_image: 
    :return:
    """
    rows, cols, channels = icon_image.shape
    image_roi = image[0:rows, 0:cols]

    img2gray = cv.cvtColor(icon_image, cv.COLOR_BGR2GRAY)
    # 高亮区域mask
    ret, mask = cv.threshold(img2gray, 50, 255, cv.THRESH_BINARY)
    # 低亮区域mask
    mask_inv = cv.bitwise_not(mask)
    # 用图标的高亮区域与图像的低亮区域组合
    img1_bg = cv.bitwise_and(image_roi, image_roi, mask=mask_inv)
    img2_fg = cv.bitwise_and(icon_image, icon_image, mask=mask)
    dst = cv.add(img1_bg, img2_fg)
    image[0:rows, 0:cols] = dst
    return image

    图片颜色网格化

    def color_reduce(im, div_power):
    """
    颜色二进制数值 末尾div_power位置 改为1
    :param im:
    :param div_power: 0<div<8
    :return:
    """
    mask1 = 0XFF << div_power
    mask2 = 0X1 << div_power-1
    rows = im.shape[0]
    cols = im.shape[1]
    channels = im.shape[2]
    for i in range(rows):
        for j in range(cols):
            for channel in range(channels):
                im[i, j][channel] = im[i, j][channel] & mask1 | mask2

    马赛克效果

    def grid_merge(img, G):
    """
    马赛克效果
    :param im:
    :param G: 马赛克格子大小
    :return:
    """
    out = img.copy()
    H, W, C = img.shape
    Nh = int(H/G) + int(H%G!=0)
    Nw = int(W/G) + int(W%G!=0)
    for y in range(Nh):
        for x in range(Nw):
            for c in range(C):
                out[G*y:G*(y+1), G*x:G*(x+1), c] = np.mean(out[G*y:G*(y+1), G*x:G*(x+1), c]).astype(np.int)
    return out

    LUT查找表使用练习

    def standardization_with_lut(gray_image):
    hist1 = cv2.calcHist([gray_image], [0], None, [256], [0.0,255.0])  
    cdf = hist1.cumsum() # 获得累加数组, 如果参数多维, 通过指定0,1,2可以确定从第几个括号累加
    cdf_m = np.ma.masked_equal(cdf,0) #除去直方图中的0值, 返回masked_array, 是带mask的数据
    lut_m = (cdf_m - cdf_m.min())*255/(cdf_m.max()-cdf_m.min())
    lut = np.ma.filled(lut_m,0).astype('uint8') #将掩模处理掉的元素补为0
    new_gray_image = cv2.LUT(img, lut)
    return new_gray_image
